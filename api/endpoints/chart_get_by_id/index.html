Get Chart By ID

根据 Chart ID 获取指定 Chart。

get
/api/v1/charts/{id}

Headers

x-datadata-api-tokenstringrequired

API Token for the service

Path Parameters

idstringrequired

Chart ID

示例代码​

  • Curl
  • Go
  • Node
  • Python
  • Java
  • Php
  • Http
curl --request GET \
  --url https://www.datadata.cn/api/v1/charts/:id \
  --header 'x-datadata-api-token: <your-token>'
package main

import (
	"fmt"
	"net/http"
	"io"
)

func main() {

	url := "https://www.datadata.cn/api/v1/charts/:id"

	req, _ := http.NewRequest("GET", url, nil)

	req.Header.Add("x-datadata-api-token", "<your-token>")

	res, _ := http.DefaultClient.Do(req)

	defer res.Body.Close()
	body, _ := io.ReadAll(res.Body)

	fmt.Println(res)
	fmt.Println(string(body))

}
const fetch = require('node-fetch');

const url = 'https://www.datadata.cn/api/v1/charts/:id';
const options = {method: 'GET', headers: {'x-datadata-api-token': '<your-token>'}};

try {
  const response = await fetch(url, options);
  const data = await response.json();
  console.log(data);
} catch (error) {
  console.error(error);
}
import http.client

conn = http.client.HTTPSConnection("www.datadata.cn")

headers = { 'x-datadata-api-token': "<your-token>" }

conn.request("GET", "/api/v1/charts/:id", headers=headers)

res = conn.getresponse()
data = res.read()

print(data.decode("utf-8"))
OkHttpClient client = new OkHttpClient();

Request request = new Request.Builder()
  .url("https://www.datadata.cn/api/v1/charts/:id")
  .get()
  .addHeader("x-datadata-api-token", "<your-token>")
  .build();

Response response = client.newCall(request).execute();
